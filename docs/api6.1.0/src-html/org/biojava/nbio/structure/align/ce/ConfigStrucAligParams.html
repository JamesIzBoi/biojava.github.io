021package org.biojava.nbio.structure.align.ce;
022
023
024import java.util.List;
025
026public interface ConfigStrucAligParams {
027
028
029        /** get the list of parameters that the user can change through the user interface.
030         *  Parameter names are the same names as the corresponding Get/Set methods.
031         *
032         * @return list of parameters
033         */
034        public List<String> getUserConfigParameters();
035
036        /** The labels to be displayed to the user for each parameter
037         *
038         * @return list of parameter names
039         */
040        public List<String> getUserConfigParameterNames();
041
042        /** Get the data types of the parameters
043         *
044         * @return list of parameter classes
045         */
046        @SuppressWarnings("rawtypes")
047        public List<Class> getUserConfigTypes();
048
049
050        /** The help text for each of these parameters.
051         *
052         * @return help strings
053         */
054        public List<String> getUserConfigHelp();
055
056
057        /** Set the parameters to the default.
058         *
059         */
060        public void reset();
061}
