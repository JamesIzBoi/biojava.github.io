021package org.biojava.nbio.structure.domain.pdp;
022
023import java.util.Arrays;
024
025
026public class CutSites {
027        int ncuts;
028        int[] cut_sites;
029
030        public CutSites(){
031                ncuts = 0;
032
033                cut_sites = new int[PDPParameters.MAX_CUTS];
034        }
035
036        @Override
037        public String toString() {
038                return "CutSites [ncuts=" + ncuts + ", cut_sites="
039                                + Arrays.toString(cut_sites) + "]";
040        }
041        public int getNcuts() {
042                return ncuts;
043        }
044        public void setNcuts(int ncuts) {
045                this.ncuts = ncuts;
046        }
047        public int[] getCut_sites() {
048                return cut_sites;
049        }
050        public void setCut_sites(int[] cut_sites) {
051                this.cut_sites = cut_sites;
052        }
053
054
055
056}
