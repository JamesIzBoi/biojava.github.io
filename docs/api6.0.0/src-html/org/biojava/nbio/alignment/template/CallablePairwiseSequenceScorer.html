024package org.biojava.nbio.alignment.template;
025
026import org.biojava.nbio.core.sequence.template.Compound;
027import org.biojava.nbio.core.sequence.template.Sequence;
028
029import java.util.concurrent.Callable;
030
031/**
032 * Implements a concurrency wrapper for a {@link PairwiseSequenceScorer}.
033 *
034 * @author Mark Chapman
035 * @param <S> each {@link Sequence} of the pair is of type S
036 * @param <C> each element of a {@link Sequence} is a {@link Compound} of type C
037 */
038public class CallablePairwiseSequenceScorer<S extends Sequence<C>, C extends Compound> implements Callable<Double> {
039
040        private PairwiseSequenceScorer<S, C> pss;
041
042        /**
043         * Creates a pairwise sequence scoring task for simplified parallel execution.
044         *
045         * @param pss already initialized pairwise sequence scorer
046         */
047        public CallablePairwiseSequenceScorer(PairwiseSequenceScorer<S, C> pss) {
048                this.pss = pss;
049        }
050
051        @Override
052        public Double call() {
053                return pss.getScore();
054        }
055
056}
