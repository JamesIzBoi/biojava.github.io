025package org.biojava.nbio.structure.xtal.io;
026
027import javax.xml.bind.annotation.adapters.XmlAdapter;
028import java.util.ArrayList;
029import java.util.List;
030
031public class TransfAlgebraicAdapter extends XmlAdapter<String[], List<String>>{
032
033        @Override
034        public String[] marshal(List<String> arg0) throws Exception {
035                String[] elements = new String[arg0.size()];
036                int i = 0;
037                for (String s : arg0)
038                        elements[i++] = s;
039                return elements;
040        }
041
042        @Override
043        public List<String> unmarshal(String[] arg0) throws Exception {
044                List<String> l = new ArrayList<String>();
045                for (String s : arg0)
046                        l.add(s);
047                return l;
048        }
049
050}
