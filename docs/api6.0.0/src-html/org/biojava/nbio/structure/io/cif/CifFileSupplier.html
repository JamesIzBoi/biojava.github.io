001package org.biojava.nbio.structure.io.cif;
002
003import org.rcsb.cif.model.CifFile;
004
005/**
006 * Create a CifFile instance for a given container of structure data.
007 * @param <S> the container type used as source
008 * @author Sebastian Bittrich
009 * @since 5.3.0
010 */
011public interface CifFileSupplier<S> {
012    /**
013     * Convert some model instance describing structure information to a CifFile instance.
014     * @param container the source of structure information
015     * @return a flat CifFile instance, ready for IO operations
016     */
017    CifFile get(S container);
018}
