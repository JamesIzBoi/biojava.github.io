021package demo;
022
023import org.biojava.nbio.core.alignment.matrices.SubstitutionMatrixHelper;
024import org.biojava.nbio.core.alignment.template.SubstitutionMatrix;
025import org.biojava.nbio.core.sequence.compound.AminoAcidCompound;
026
027public class DemoLoadSubstMax {
028
029
030        public static void main (String[] args){
031                // that's the PAM250 matrix (named a bit unclear in AAindex...)
032                String max2="DAYM780301";
033                SubstitutionMatrix<AminoAcidCompound> substMax2 = SubstitutionMatrixHelper.getMatrixFromAAINDEX(max2);
034                System.out.println("PAM250 matrix: "+ substMax2);
035
036                // and here BLOSUM62...
037                String max3="HENS920102";
038                SubstitutionMatrix<AminoAcidCompound> substMax3 = SubstitutionMatrixHelper.getMatrixFromAAINDEX(max3);
039                System.out.printf("%s matrix: %s", max3, substMax3);
040                System.out.println();
041
042                // This one I developed a while ago to  be optimised for the alignment of distantly related sequences
043                String matrixName4 = "PRLA000101";
044                SubstitutionMatrix<AminoAcidCompound> substMax4 = SubstitutionMatrixHelper.getMatrixFromAAINDEX(matrixName4);
045                System.out.printf("%s matrix: %s", matrixName4, substMax4);
046                System.out.println();
047        }
048}
